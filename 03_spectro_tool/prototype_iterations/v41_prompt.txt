Response
========

The bubbles disappeared because the current implementation combines:

```js
const fade =
    Math.max(0, 1 - dist / 160);
```

with:

```js
rgba(..., 0.18 * fade)
```

Most bubbles end up rendered at extremely low alpha (effectively invisible), especially when:

* the mouse is not nearby
* playback scrubber is elsewhere
* the row is dense

The older implementation from ~8–9 revisions ago worked better because it used:

* persistent visible inactive markers
* only one expanded active bubble
* simple distance activation
* stronger inactive alpha
* stable rounded-rect dots instead of fading nearly to zero

That version also avoided overlap/merge artifacts.

You also requested:

* F pitches = `#F5D954`

So:

```js
return [245,217,84];
```

replaces the previous yellow-green mapping.

function colorForFreq(freq){

    const midi =
        tunedMidi(freq);

    const octave =
        Math.floor(midi/12);

    const pitch =
        Math.round(midi)%12;

    switch(pitch){

        case 0:
            return [255,210,160];

        case 2:
            return [40,60,150];

        case 4:
            return (
                freq >= aTuningHz
                ? [220,180,255]
                : [120,80,50]
            );

        case 5:
            return [245,217,84];

        case 7:

            if(octave < 4)
                return [0,80,30];

            if(octave === 4)
                return [70,190,50];

            return [180,255,80];

        case 9:
            return [255,70,70];

        case 11:
            return [120,220,255];

        default:
            return [180,180,180];

}

/* ---------- RESTORED BUBBLE SYSTEM ---------- */

function drawBubbles(){

    let activeFound = false;

    for(const e of midiEvents){

        const x =
            (
                e.start /
                audioDuration
            ) *
            canvas.width;

        const dist =
            Math.abs(x - mouseX);

        const active =
            dist < 36 &&
            !activeFound;

        if(active)
            activeFound = true;

        const bubbleY =
            canvas.height * 0.84;

        if(active){

            const label =
                e.label;

            ctx.font =
                "bold 12px monospace";

            const tw =
                ctx.measureText(label)
                .width;

            const w =
                tw + 22;

            let bx =
                x - w * 0.5;

            bx =
                Math.max(
                    4,
                    Math.min(
                        canvas.width - w - 4,
                        bx
                    )
                );

            ctx.fillStyle =
                "rgba(255,140,40,0.92)";

            ctx.beginPath();

            ctx.roundRect(
                bx,
                bubbleY - 12,
                w,
                22,
                10
            );

            ctx.fill();

            ctx.fillStyle =
                "white";

            ctx.fillText(
                label,
                bx + 11,
                bubbleY + 3
            );

        }else{

            ctx.fillStyle =
                "rgba(80,120,255,0.58)";

            ctx.beginPath();

            ctx.arc(
                x,
                bubbleY,
                3.2,
                0,
                Math.PI * 2
            );

            ctx.fill();
        }
    }
}

</script>
