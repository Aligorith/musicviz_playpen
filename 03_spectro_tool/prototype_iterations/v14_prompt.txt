# 🔥 2. Add CQT frequency map

```js id="cqt_map"
function buildCQTBins() {

    const bins = [];

    const octaves =
        Math.log2(CQT_MAX_FREQ / CQT_MIN_FREQ);

    const totalBins =
        Math.floor(octaves * CQT_BINS_PER_OCTAVE);

    for (let i = 0; i < totalBins; i++) {

        const freq =
            CQT_MIN_FREQ *
            Math.pow(
                2,
                i / CQT_BINS_PER_OCTAVE
            );

        bins.push(freq);
    }

    return bins;
}

const CQT_BINS = buildCQTBins();
```

# 🔥 3. CQT kernel (log-weighted windowing)

```js id="cqt_kernel"
function cqtKernel(signal, targetFreq, sr) {

    const N = signal.length;

    const windowSize =
        Math.floor(sr / targetFreq * 2.5);

    let sum = 0;

    for (let i = 0; i < N; i++) {

        const phase =
            (2 * Math.PI * targetFreq * i) / sr;

        const window =
            0.5 * (1 - Math.cos((2 * Math.PI * i) / (N - 1)));

        sum +=
            signal[i] *
            Math.cos(phase) *
            window;
    }

    return Math.abs(sum / windowSize);
}
```

# 🔥 4. Replace FFT-based pitch detection with CQT

### NEW CORE FUNCTION:

```js id="cqt_analyze"
function analyzeCQT(frame, sampleRate) {

    const energies = [];

    let maxVal = 0;

    for (let i = 0; i < CQT_BINS.length; i++) {

        const f = CQT_BINS[i];

        const mag =
            cqtKernel(frame, f, sampleRate);

        energies.push({ freq: f, mag });

        if (mag > maxVal) maxVal = mag;
    }

    /*
        Log harmonic smoothing:
        - smooth across octaves
        - suppress isolated spikes
    */
    for (let i = 1; i < energies.length - 1; i++) {

        energies[i].mag =
            (energies[i - 1].mag +
             energies[i].mag +
             energies[i + 1].mag) / 3;
    }

    return energies
        .filter(e => e.mag > maxVal * 0.25)
        .sort((a,b) => b.mag - a.mag);
}
```

# 🔥 5. Polyphonic note tracking (important)

Replace `peaks()` usage entirely.

Add:

```js id="poly_notes"
function extractNotes(cqtFrame) {

    const notes = [];

    const threshold =
        Math.max(...cqtFrame.map(e => e.mag)) * 0.35;

    for (let i = 1; i < cqtFrame.length - 1; i++) {

        const e = cqtFrame[i];

        if (
            e.mag < threshold ||
            e.mag < cqtFrame[i - 1].mag ||
            e.mag < cqtFrame[i + 1].mag
        ) continue;

        const midi =
            freqToMidi(e.freq);

        notes.push({
            freq: e.freq,
            midi,
            label: midiToLily(midi),
            mag: e.mag
        });
    }

    return notes.slice(0, 6);
}
```

```js id="midi_add"
function recordMidi(frameIndex, notes, time) {

    for (const n of notes) {

        midiEvents.push({
            t: time,
            freq: n.freq,
            midi: n.midi,
            label: n.label,
            mag: n.mag
        });
    }
}
```

---

### Add export function:

```js id="midi_export"
function exportMidi() {

    let text = "MIDI NOTE EXPORT\n\n";

    for (const e of midiEvents) {

        text +=
            `${e.t.toFixed(2)}s  ` +
            `${e.label}  ` +
            `${e.freq.toFixed(2)}Hz\n`;
    }

    const blob =
        new Blob([text], { type: "text/plain" });

    const url =
        URL.createObjectURL(blob);

    const a =
        document.createElement("a");

    a.href = url;
    a.download = "notes.txt";
    a.click();
}
```
