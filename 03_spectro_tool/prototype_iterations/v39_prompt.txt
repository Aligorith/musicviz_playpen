<!-- ADD this helper function near playAt() -->

function previewAt(time){

    if(!audioBuffer)
        return;

    const now =
        performance.now();

    if(
        Math.abs(
            time -
            lastScrubPreviewTime
        ) < 0.03
    ){
        return;
    }

    lastScrubPreviewTime = time;

    if(scrubPreviewSource){

        try{
            scrubPreviewSource.stop();
        }catch(e){}
    }

    scrubPreviewSource =
        audioCtx.createBufferSource();

    scrubPreviewSource.buffer =
        audioBuffer;

    const gain =
        audioCtx.createGain();

    gain.gain.value = 0.28;

    scrubPreviewSource.connect(gain);

    gain.connect(
        audioCtx.destination
    );

    scrubPreviewSource.start(
        0,
        Math.max(0,time),
        0.09
    );
}
```

function exportDetectedNotes(){

    if(!midiEvents.length)
        return;

    let txt =
        "time_seconds,note,freq_hz\n";

    for(const e of midiEvents){

        txt +=
            `${e.start.toFixed(3)},${e.label},${e.freq.toFixed(2)}\n`;
    }

    const blob =
        new Blob(
            [txt],
            {
                type:"text/csv"
            }
        );

    const url =
        URL.createObjectURL(blob);

    const a =
        document.createElement("a");

    a.href = url;

    a.download =
        "detected_notes.csv";

    a.click();

    URL.revokeObjectURL(url);
}
```
