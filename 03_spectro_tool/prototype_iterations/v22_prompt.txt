Now add this entire new function BELOW `process()`:

```js id="redraw_function"
function redrawSpectrogram(){

    specCtx.clearRect(
        0,
        0,
        canvas.width,
        canvas.height
    );

    if(
        !cachedSpectrogram.length
    ) return;

    const xScale =
        canvas.width /
        cachedSpectrogram.length;

    for(
        let i = 0;
        i < cachedSpectrogram.length;
        i++
    ){

        const spec =
            cachedSpectrogram[i];

        for(
            let b = 1;
            b < spec.length;
            b++
        ){

            const freq =
                b *
                (
                    cachedSampleRate /
                    cachedFFTSize
                );

            if(
                freq < 20 ||
                freq > 8000
            ) continue;

            const y =
                canvas.height -
                40 -
                (
                    Math.log2(freq / 20) /
                    Math.log2(8000 / 20)
                ) *
                (
                    canvas.height - 180
                );

            const mag =
                spec[b];

            const alpha =
                Math.min(
                    1,
                    (
                        mag / 55
                    ) *
                    brightnessBoost
                );

            const c =
                colorForFreq(freq);

            specCtx.fillStyle =
                `rgba(
                    ${c[0]},
                    ${c[1]},
                    ${c[2]},
                    ${alpha}
                )`;

            specCtx.fillRect(
                i * xScale,
                y,
                Math.ceil(xScale),
                1
            );
        }
    }
}
```
