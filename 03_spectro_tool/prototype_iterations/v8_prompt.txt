/*
    Pitch colors

const NOTE_COLORS = {

    c: [255,210,150],
    d: [30,40,120],

    e_high: [220,180,255],
    e_low: [110,70,40],

    f: [190,255,70],

    g: [0,255,0],

    a: [255,60,60],

    b: [120,220,255]
};

const NOTE_NAMES = [

    "c","cis","d","ees","e","f",
    "fis","g","gis","a","bes","b"
];

function freqToMidi(freq) {

    return (
        69 +
        12 * Math.log2(freq / 440)
    );
}

function midiToLily(midi) {

    const rounded =
        Math.round(midi);

    const note =
        NOTE_NAMES[
            ((rounded % 12) + 12) % 12
        ];

    const octave =
        Math.floor(rounded / 12) - 1;

    let suffix = "";

    if (octave >= 3) {
        suffix =
            "'".repeat(octave - 3);
    }
    else {
        suffix =
            ",".repeat(3 - octave);
    }

    return note + suffix;
}

function formatTime(seconds) {

    seconds = Math.max(0, seconds);

    const mins =
        Math.floor(seconds / 60);

    const secs =
        Math.floor(seconds % 60);

    return (
        String(mins).padStart(2,"0") +
        ":" +
        String(secs).padStart(2,"0")
    );
}

function getPitchClass(midi) {

    const pc =
        ((Math.round(midi) % 12) + 12) % 12;

    const names = [

        "c","c","d","d","e","f",
        "f","g","g","a","a","b"
    ];

    return names[pc];
}

function getPitchColor(freq) {

    const midi =
        freqToMidi(freq);

    const octave =
        Math.floor(midi / 12);

    const pc =
        getPitchClass(midi);

    let rgb;

    switch(pc) {

        case "c":
            rgb = NOTE_COLORS.c;
            break;

        case "d":
            rgb = NOTE_COLORS.d;
            break;

        case "e":
            rgb =
                freq >= 440
                ? NOTE_COLORS.e_high
                : NOTE_COLORS.e_low;
            break;

        case "f":
            rgb = NOTE_COLORS.f;
            break;

        case "g":

            if (octave < 4) {
                rgb = [0,90,20];
            }
            else if (octave === 4) {
                rgb = [80,200,50];
            }
            else {
                rgb = [160,255,60];
            }

            break;

        case "a":
            rgb = NOTE_COLORS.a;
            break;

        case "b":
            rgb = NOTE_COLORS.b;
            break;

        default:
            rgb = [180,180,180];
    }

    const brightness =
        Math.max(
            0.25,
            Math.min(1.2, octave / 6)
        );

    return rgb.map(v =>
        Math.max(
            0,
            Math.min(255, v * brightness)
        )
    );
}

function naiveFFT(signal) {

    const N = signal.length;

    const spectrum =
        new Float32Array(N / 2);

    for (let k = 0; k < N / 2; k++) {

        let re = 0;
        let im = 0;

        for (let n = 0; n < N; n++) {

            const phi =
                (2 * Math.PI * k * n) / N;

            re +=
                signal[n] * Math.cos(phi);

            im -=
                signal[n] * Math.sin(phi);
        }

        spectrum[k] =
            Math.sqrt(re*re + im*im);
    }

    return spectrum;
}

async function decodeFile(file) {

    const buffer =
        await file.arrayBuffer();

    return await
        audioCtx.decodeAudioData(buffer);
}

function analyzeDominantPitches(
    spectrum,
    sampleRate,
    fftSize
) {

    const peaks = [];

    for (
        let i = 1;
        i < spectrum.length - 1;
        i++
    ) {

        if (
            spectrum[i] > spectrum[i - 1] &&
            spectrum[i] > spectrum[i + 1]
        ) {

            peaks.push({

                freq:
                    i *
                    sampleRate /
                    fftSize,

                mag:
                    spectrum[i]
            });
        }
    }

    peaks.sort((a,b) => b.mag - a.mag);

    return peaks
        .slice(0, 6)
        .map(p => ({

            label:
                midiToLily(
                    freqToMidi(p.freq)
                ),

            mag: p.mag
        }));
}

async function generateSpectrogram(audioBuffer) {

    bubbles.length = 0;

    spectrogramReady = false;

    audioDuration =
        audioBuffer.duration;

    const width =
        spectrogramCanvas.width;

    const height =
        spectrogramCanvas.height;

    spectrogramCtx.fillStyle = "#000";

    spectrogramCtx.fillRect(
        0,
        0,
        width,
        height
    );

    const data =
        audioBuffer.getChannelData(0);

    const fftSize = 1024;
    const hop = 512;

    const totalFrames =
        Math.floor(
            (data.length - fftSize) / hop
        );

    const xScale =
        width / totalFrames;

    let previousLabels = "";

    for (
        let frame = 0;
        frame < totalFrames;
        frame++
    ) {

        if (frame % 5 === 0) {

            statusEl.textContent =
                `Analyzing ${frame}/${totalFrames}`;

            await new Promise(r => setTimeout(r,0));
        }

        const start =
            frame * hop;

        const segment =
            data.slice(
                start,
                start + fftSize
            );

        for (let i = 0; i < segment.length; i++) {

            segment[i] *=
                0.5 *
                (
                    1 -
                    Math.cos(
                        (
                            2 *
                            Math.PI *
                            i
                        ) /
                        (segment.length - 1)
                    )
                );
        }

        const spectrum =
            naiveFFT(segment);

        /*
            Spectrogram below UI
        */
        for (
            let bin = 0;
            bin < spectrum.length;
            bin++
        ) {

            const freq =
                bin *
                audioBuffer.sampleRate /
                fftSize;

            if (
                freq < 20 ||
                freq > 8000
            ) continue;

            const mag =
                spectrum[bin];

            const y =
                uiOffsetY +
                (
                    height - uiOffsetY
                ) -
                (
                    Math.log2(freq / 20) /
                    Math.log2(8000 / 20)
                ) *
                (
                    height - uiOffsetY
                );

            const color =
                getPitchColor(freq);

            const alpha =
                Math.min(1, mag / 60);

            spectrogramCtx.fillStyle =
                `rgba(
                    ${color[0]},
                    ${color[1]},
                    ${color[2]},
                    ${alpha}
                )`;

            spectrogramCtx.fillRect(
                frame * xScale,
                y,
                Math.ceil(xScale),
                1
            );
        }

        const dominant =
            analyzeDominantPitches(
                spectrum,
                audioBuffer.sampleRate,
                fftSize
            );

        const labels =
            dominant
                .map(d => d.label)
                .join(" ");

        if (labels !== previousLabels) {

            previousLabels = labels;

            bubbles.push({

                x:
                    frame * xScale,

                y:
                    uiOffsetY +
                    (
                        canvas.height -
                        uiOffsetY
                    ) * 0.72,

                labels,

                dominantLabel:
                    dominant.length
                    ? dominant[0].label
                    : "",

                active: false,

                opacity: 0,
                targetOpacity: 0,

                currentWidth: 10
            });
        }
    }

    spectrogramReady = true;

    statusEl.textContent =
        "Analysis complete.";
}
