* `analyzeDominantPitches`

### REPLACE WITH THIS:

```js id="hps_pitch"
function harmonicProductSpectrum(spectrum) {

    const N = spectrum.length;
    const hps = new Float32Array(N);

    for (let i = 0; i < N; i++) {
        hps[i] = spectrum[i];
    }

    for (let h = 2; h <= 5; h++) {
        for (let i = 0; i < N / h; i++) {
            hps[i] *= spectrum[i * h];
        }
    }

    return hps;
}
```

# 🔧 4. Parabolic peak refinement

Add:

```js id="parab"
function parabolicInterp(mag, i) {

    const alpha = mag[i - 1];
    const beta  = mag[i];
    const gamma = mag[i + 1];

    const denom =
        alpha - 2 * beta + gamma;

    if (denom === 0) return i;

    const offset =
        0.5 * (alpha - gamma) / denom;

    return i + offset;
}
```

# 🔧 5. Replace `peaks()` function entirely

```js id="new_peaks"
function peaks(spectrum, sr, fftSize) {

    const hps = harmonicProductSpectrum(spectrum);

    const maxVal =
        Math.max(...hps);

    const results = [];

    for (let i = 2; i < hps.length - 2; i++) {

        const val = hps[i];

        if (val < maxVal * 0.15) continue;

        if (
            val > hps[i - 1] &&
            val > hps[i + 1]
        ) {

            const refined =
                parabolicInterp(hps, i);

            const freq =
                refined * sr / fftSize;

            if (freq < 60 || freq > 2000) continue;

            results.push({
                freq,
                mag: val
            });
        }
    }

    results.sort((a,b) => b.mag - a.mag);

    return results.slice(0, 6)
        .map(p => ({
            label: midiToLily(freqToMidi(p.freq)),
            mag: p.mag
        }));
}
```
