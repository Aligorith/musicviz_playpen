Prompt
======

* On the scrubber, include markings for each "c" on the scale
* When click-dragging the scrubber, allow playing back the part of the soundfile passing under the cursor
* In the scrubber bubble, show a small rounded rect of the note color associated with the pitch it is naming
* The lower end on the spectrogram doesn't need to go below the g below the g-below-middle-c

function dominantPitchAtTime(t){

    let best = null;

    for(const e of midiEvents){

        if(
            t >= e.start &&
            t <= e.end
        ){

            if(
                !best ||
                e.freq > best.freq
            ){
                best = e;
            }
        }
    }

    return best;
}

Now ADD this entire function BELOW `drawTimeline()`:

```js id="draw_c_markers"
function drawCMarkers(){

    const cFreqs = [];

    for(let o=0;o<9;o++){

        const midi =
            12 + o*12;

        const freq =
            440 *
            Math.pow(
                2,
                (midi - 69)/12
            );

        if(
            freq >= MIN_FREQ &&
            freq <= MAX_FREQ
        ){
            cFreqs.push({
                freq,
                octave:o
            });
        }
    }

    for(const c of cFreqs){

        const y =
            canvas.height -
            40 -
            (
                Math.log2(
                    c.freq / MIN_FREQ
                ) /
                Math.log2(
                    MAX_FREQ / MIN_FREQ
                )
            ) *
            (
                canvas.height - 180
            );

        ctx.strokeStyle =
            "rgba(255,255,255,0.25)";

        ctx.beginPath();

        ctx.moveTo(0,y);
        ctx.lineTo(14,y);

        ctx.stroke();

        ctx.fillStyle =
            "rgba(255,255,255,0.65)";

        ctx.font =
            "11px monospace";

        ctx.fillText(
            `c${c.octave}`,
            18,
            y + 4
        );
    }
}
```

Finally, add this entire interaction block near the bottom with the other mouse interaction handlers:

```js id="audio_scrub_playback"
function stopPlayback(){

    if(playbackSource){

        try{
            playbackSource.stop();
        }catch(e){}

        playbackSource.disconnect();

        playbackSource = null;
    }
}

function playPreviewAt(x){

    if(!audioBuffer)
        return;

    stopPlayback();

    const t =
        (
            x / canvas.width
        ) * audioDuration;

    const src =
        audioCtx.createBufferSource();

    src.buffer = audioBuffer;

    src.connect(
        audioCtx.destination
    );

    src.start(
        0,
        Math.max(0,t),
        0.15
    );

    playbackSource = src;
}

canvas.addEventListener(
    "mousedown",
    e => {

        scrubbingAudio = true;

        mouseX = e.clientX;

        playPreviewAt(mouseX);
    }
);

canvas.addEventListener(
    "mousemove",
    e => {

        mouseX = e.clientX;

        if(scrubbingAudio){

            playPreviewAt(mouseX);
        }
    }
);

addEventListener(
    "mouseup",
    () => {

        scrubbingAudio = false;

        stopPlayback();
    }
);
```

This gives:

* audible scrub-preview playback
* colored note chip in scrubber bubble
* visible "C" pitch guide markings
* reduced low-frequency clutter
* proper musical lower range floor
* DAW-style click-drag scrub behavior
